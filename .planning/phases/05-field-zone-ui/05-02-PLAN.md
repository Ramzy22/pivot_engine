"""
Phase 5 regression tests — field zone UI
Covers: server-side min/max aggregation (FIELD-03), prop round-trip sanity (FIELD-05, FIELD-06)
"""
import pytest
import sys
import os
import pyarrow as pa

sys.path.append(os.getcwd())
sys.path.append(os.path.join(os.getcwd(), 'pivot_engine'))

from pivot_engine.tanstack_adapter import create_tanstack_adapter, TanStackRequest, TanStackOperation


@pytest.fixture
def adapter():
    a = create_tanstack_adapter(backend_uri=":memory:")
    table = pa.table({
        "region": ["North", "North", "South", "South"],
        "sales": [100, 200, 300, 50],
    })
    a.controller.load_data_from_arrow("test", table)
    return a


@pytest.mark.asyncio
async def test_min_aggregation_is_server_side(adapter):
    """agg='min' must return minimum sales per region, not a sum or incorrect value."""
    req = TanStackRequest(
        operation=TanStackOperation.INITIAL_LOAD,
        pivot_spec={
            "table": "test",
            "rows": ["region"],
            "cols": [],
            "measures": [{"field": "sales", "agg": "min"}],
        },
        viewport={"startRow": 0, "endRow": 50},
    )
    result = await adapter.handle_request(req)
    rows = result.get("rows", [])
    # Find North row
    north = next((r for r in rows if r.get("region") == "North" or r.get("_id") == "North"), None)
    assert north is not None, f"North row not found in {rows}"
    min_key = next((k for k in north if "sales" in k and "min" in k), None)
    assert min_key is not None, f"No min sales key in {north}"
    assert north[min_key] == 100, f"Expected min=100 for North, got {north[min_key]}"


@pytest.mark.asyncio
async def test_max_aggregation_is_server_side(adapter):
    """agg='max' must return maximum sales per region."""
    req = TanStackRequest(
        operation=TanStackOperation.INITIAL_LOAD,
        pivot_spec={
            "table": "test",
            "rows": ["region"],
            "cols": [],
            "measures": [{"field": "sales", "agg": "max"}],
        },
        viewport={"startRow": 0, "endRow": 50},
    )
    result = await adapter.handle_request(req)
    rows = result.get("rows", [])
    south = next((r for r in rows if r.get("region") == "South" or r.get("_id") == "South"), None)
    assert south is not None, f"South row not found in {rows}"
    max_key = next((k for k in south if "sales" in k and "max" in k), None)
    assert max_key is not None, f"No max sales key in {south}"
    assert south[max_key] == 300, f"Expected max=300 for South, got {south[max_key]}"


@pytest.mark.asyncio
async def test_initial_rowfields_used_in_request(adapter):
    """rowFields passed as pivot_spec rows must produce grouped output (FIELD-06 server side)."""
    req = TanStackRequest(
        operation=TanStackOperation.INITIAL_LOAD,
        pivot_spec={
            "table": "test",
            "rows": ["region"],
            "cols": [],
            "measures": [{"field": "sales", "agg": "sum"}],
        },
        viewport={"startRow": 0, "endRow": 50},
    )
    result = await adapter.handle_request(req)
    rows = result.get("rows", [])
    ids = [r.get("region") or r.get("_id") for r in rows if not r.get("_isTotal")]
    assert "North" in ids, f"Expected 'North' in row ids: {ids}"
    assert "South" in ids, f"Expected 'South' in row ids: {ids}"
